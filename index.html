        function renderMockup() {

        /**

         * 渲染左側解說面板的內容，與當前 UI 截圖同步。

         */

        function renderInfoPanel() {

            const slide = activeProject.slides[activeSlideIndex];

            

            document.getElementById('project-name').textContent = activeProject.name;

            document.getElementById('slide-zh-title').textContent = slide.zh_title;

            document.getElementById('slide-en-title').textContent = slide.en_title;

            document.getElementById('slide-zh-desc').textContent = slide.zh_desc;

            document.getElementById('slide-en-desc').textContent = slide.en_desc;

        }


        /**

         * 渲染下方專案縮圖。

         */

        function renderThumbnails() {

            const container = document.getElementById('project-thumbnails');

            container.innerHTML = '';


            projects.forEach(project => {

                const isActive = project.id === activeProject.id;

                const thumbnail = document.createElement('div');

                

                thumbnail.className = `flex flex-col items-center cursor-pointer p-2 rounded-lg transition transform hover:scale-105 ${isActive ? 'ring-4 ring-indigo-500 bg-indigo-50' : 'bg-white hover:bg-gray-100'}`;

                thumbnail.onclick = () => switchProject(project.id);


                const image = project.slides[0] ? project.slides[0].image : 'https://placehold.co/100x200/cccccc/000000?text=No+Image';

                

                thumbnail.innerHTML = `

                    <div class="w-16 h-32 overflow-hidden rounded-xl border-4 border-gray-800 shadow-md">

                        <img src="${image}" alt="${project.name}" class="w-full h-full object-cover">

                    </div>

                    <p class="mt-2 text-center text-xs font-medium ${isActive ? 'text-indigo-700' : 'text-gray-700'}">${project.name.split(' ')[0]}</p>

                `;

                container.appendChild(thumbnail);

            });

        }

        }


        /**

         * 切換到下一個 UI 頁面。

        function nextSlide() {

        /**

         * 切換活動專案。

         * @param {string} projectId - 要切換到的專案 ID。

         */

        function switchProject(projectId) {

            const newProject = projects.find(p => p.id === projectId);

            if (newProject) {

                activeProject = newProject;

                activeSlideIndex = 0; // 重置頁面索引

                renderMockup();

                renderInfoPanel();

                renderThumbnails(); // 重新渲染以更新 active 狀態

                document.getElementById('active-mockup-container').classList.remove('hidden'); // 確保裝置顯示

            }

        }


        /**

         * 顯示專案詳情 Modal。

         */

        function showCaseStudy() {

            renderCaseStudy();

            document.getElementById('case-study-modal').classList.remove('hidden');

            document.body.style.overflow = 'hidden'; // 禁用背景滾動

        }


        /**

         * 隱藏專案詳情 Modal。

         */

        function hideCaseStudy() {

            document.getElementById('case-study-modal').classList.add('hidden');

            document.body.style.overflow = 'auto'; // 啟用背景滾動

        }


        /**

         * 切換專案詳情 Modal 的語言。

         * @param {string} lang - 'zh' 或 'en'。

         */

        function switchLanguage(lang) {

            activeLanguage = lang;

            renderCaseStudy();


            // 更新按鈕樣式

            document.getElementById('lang-zh').className = activeLanguage === 'zh' 

                ? 'px-4 py-2 text-white font-bold rounded-lg bg-indigo-600'

                : 'px-4 py-2 text-indigo-600 font-bold rounded-lg border border-indigo-600 hover:bg-indigo-50 transition';

            

            document.getElementById('lang-en').className = activeLanguage === 'en' 

                ? 'px-4 py-2 text-white font-bold rounded-lg bg-indigo-600'

                : 'px-4 py-2 text-indigo-600 font-bold rounded-lg border border-indigo-600 hover:bg-indigo-50 transition';

        }



        /**

         * 渲染專案詳情 Modal 的內容。

         */

        function renderCaseStudy() {

            const contentDiv = document.getElementById('case-study-content');

            const data = activeProject.caseStudy[activeLanguage];

            const isZh = activeLanguage === 'zh';

            

            document.getElementById('cs-title').textContent = activeProject.name;

            

            const sections = [

                { key: 'background', zh_title: '1. 背景', en_title: '1. Background' },

                { key: 'painPoints', zh_title: '2. 痛點', en_title: '2. Pain Points' },

                { key: 'solutions', zh_title: '3. 設計解決方案', en_title: '3. Design Solutions' },

                { key: 'conclusion', zh_title: '4. 結論', en_title: '4. Conclusion' },

            ];


            let htmlContent = '';


            sections.forEach(section => {

                const title = isZh ? section.zh_title : section.en_title;

                const body = data[section.key] || '內容待補 (Content pending)';

                

                htmlContent += `

                    <div class="case-study-section">

                        <h3 class="text-2xl font-bold text-gray-800 mb-3 border-l-4 border-indigo-500 pl-3">

                            ${title}

                        </h3>

                        <p class="text-gray-700 leading-loose">

                            ${body}

                        </p>

                    </div>

                `;

                

                // 特別在 "設計解決方案" 之後插入 UI 截圖

                if (section.key === 'solutions' && activeProject.slides.length > 0) {

                    htmlContent += `

                        <div class="flex flex-wrap justify-center gap-6 p-4 bg-gray-50 rounded-xl shadow-inner">

                            <h4 class="w-full text-center text-xl font-semibold text-indigo-700 mb-4">

                                關鍵 UI 截圖預覽 (Key UI Screenshots Preview)

                            </h4>

                            ${activeProject.slides.map(slide => `

                                <div class="w-32 md:w-40 bg-white p-2 rounded-xl shadow-lg transform hover:scale-105 transition duration-300">

                                    <img src="${slide.image}" alt="${slide.zh_title}" class="w-full h-auto rounded-lg border border-gray-200">

                                    <p class="text-center text-xs mt-1 text-gray-600 truncate">${slide.zh_title}</p>

                                </div>

                            `).join('')}

                        </div>

                    `;

                }

            });


            contentDiv.innerHTML = htmlContent;

        }


        // 啟動應用程式

        window.onload = initApp;

    </script>
